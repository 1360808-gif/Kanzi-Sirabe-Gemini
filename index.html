async function getReading(word) { try { // 文末の句読点などが混じると精度が落ちるため、漢字と数字のみに絞る const cleanWord = word.trim(); const url = `https://translate.googleapis.com/translate_a/single?client=gtx&sl=ja&tl=ja&dt=rm&q=${encodeURIComponent(cleanWord)}`; const response = await fetch(url); const data = await response.json(); // Google翻訳APIからローマ字を取得 let romaji = (data[0] && data[0][0] && data[0][0][3]) ? data[0][0][3] : ""; if (!romaji) return ""; // 特殊記号や長音の正規化（精度の向上） let fixed = romaji.toLowerCase() .replace(/[āâ]/g, 'aa') .replace(/[īî]/g, 'ii') .replace(/[ūû]/g, 'uu') .replace(/[ēê]/g, 'ei') .replace(/[ōô]/g, 'ou') .replace(/[^\w\s]/g, ''); // 記号を削除 return wanakana.toHiragana(fixed); } catch (e) { console.error("読みの取得に失敗しました", e); return ""; } }
